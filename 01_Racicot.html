Population relative en prison travers les annes

prison %>%
  group_by(state, year) %>%
  mutate(relative_jail_pop = total_jail_pop/total_pop*100) %>%
  summarise(med_pop = median(relative_jail_pop)) %>%
  ggplot(aes(x=year, y=med_pop, colour=state)) +
    geom_line()
